前端：
frontend/
├── public/                 # 静态资源
│   └── index.html
├── src/
│   ├── assets/            # 资源文件（图片、样式等）
│   ├── components/        # 可复用组件
│   │   ├── Common/       # 通用组件（如按钮、输入框）
│   │   ├── Auth/         # 认证相关（登录/注册表单）
│   │   └── Conversation/ # 对话界面组件
│   ├── views/            # 页面组件
│   │   ├── LoginView.vue    # 登录页
│   │   ├── RegisterView.vue # 注册页
│   │   ├── DashboardView.vue # 主控制台
│   │   └── ConversationView.vue # 对话管理页
│   ├── router/           # 路由配置
│   │   └── index.js      # 定义路由（如登录、对话历史）
│   ├── store/            # 状态管理（Pinia）
│   │   └── auth.js       # 用户认证状态
│   ├── utils/            # 工具函数
│   │   └── api.js        # API调用封装（统一处理请求）
│   ├── api/              # API服务模块
│   │   ├── auth.js       # 认证API（登录、注册）
│   │   └── conversation.js # 对话API
│   ├── App.vue           # 根组件
│   └── main.js           # 应用入口
├── package.json
├── vite.config.js        # Vite配置
└── .env                  # 环境变量（如后端API地址）


后端：
backend/
├── app/
│   ├── __init__.py
│   ├── main.py           # FastAPI应用入口
│   ├── models/           # SQLAlchemy模型（对应数据库表）
│   │   ├── __init__.py
│   │   ├── user.py       # UserStorage模型
│   │   └── conversation.py # ConversationStorage模型
│   ├── schemas/          # Pydantic模式（请求/响应验证）
│   │   ├── user.py       # 用户模式（如UserCreate、UserResponse）
│   │   └── conversation.py
│   ├── routes/           # API路由
│   │   ├── __init__.py
│   │   ├── auth.py       # 认证路由（/login, /register）
│   │   └── conversation.py # 对话路由（/conversations）
│   ├── database.py       # 数据库连接和会话管理
│   └── config.py         # 应用配置（数据库URL等）
├── requirements.txt      # Python依赖
└── Dockerfile           # 容器化配置
